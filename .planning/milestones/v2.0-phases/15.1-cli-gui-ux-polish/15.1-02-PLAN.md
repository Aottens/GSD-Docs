Create three new hook files. Each follows the same pattern:
1. Call the original polling hook
2. Track previous data fingerprint in a `useRef<string | undefined>(undefined)`
3. In a `useEffect([query.data])`, compute current fingerprint
4. If `prevRef.current === undefined` (first load), set ref silently and return (NO toast)
5. If `prevRef.current !== curr`, fire `toast.info()` and update ref
6. Return the original query object unchanged

**File 1: `frontend/src/features/projects/hooks/useSetupStateWithNotification.ts`**

```typescript
import { useRef, useEffect } from 'react'
import { toast } from 'sonner'
import { useSetupState } from './useSetupState'
import type { SetupStateResponse } from '@/types/setupState'

function fingerprint(data: SetupStateResponse): string {
  return data.doc_types?.map(d => `${d.id}:${d.status}:${d.file_count}`).join(',') ?? ''
}

export function useSetupStateWithNotification(projectId: number) {
  const query = useSetupState(projectId)
  const prevRef = useRef<string | undefined>(undefined)

  useEffect(() => {
    if (!query.data) return
    const curr = fingerprint(query.data)
    if (prevRef.current === undefined) {
      prevRef.current = curr
      return
    }
    if (prevRef.current !== curr) {
      toast.info('Setup status bijgewerkt', {
        description: 'CLI heeft wijzigingen aangebracht in de projectconfiguratie.',
      })
      prevRef.current = curr
    }
  }, [query.data])

  return query
}
```

**File 3: `frontend/src/features/documents/hooks/useDocumentOutlineWithNotification.ts`**

```typescript
import { useRef, useEffect } from 'react'
import { toast } from 'sonner'
import { useDocumentOutline } from './useDocumentOutline'
import type { DocumentOutlineResponse, OutlineNode } from '../types/document'

function countWithContent(nodes: OutlineNode[]): number {
  return nodes.reduce((acc, n) => {
    const self = n.status === 'written' || n.status === 'verified' ? 1 : 0
    return acc + self + countWithContent(n.children ?? [])
  }, 0)
}

function fingerprint(data: DocumentOutlineResponse): string {
  return `${data.sections.length}:${countWithContent(data.sections)}`
}

export function useDocumentOutlineWithNotification(projectId: number) {
  const query = useDocumentOutline(projectId)
  const prevRef = useRef<string | undefined>(undefined)

  useEffect(() => {
    if (!query.data) return
    const curr = fingerprint(query.data)
    if (prevRef.current === undefined) {
      prevRef.current = curr
      return
    }
    if (prevRef.current !== curr) {
      toast.info('Documentstructuur bijgewerkt', {
        description: 'Nieuwe secties beschikbaar na CLI schrijffase.',
      })
      prevRef.current = curr
    }
  }, [query.data])

  return query
}
```
