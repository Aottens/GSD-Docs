```typescript
export function useAssemblyStream(projectId: number) {
  // State: stages (PipelineStage[]), isRunning (boolean)
  // eventSourceRef for cleanup
  // Initial stages: 4 items, all 'idle':
  //   { step: 0, name: 'Cross-referenties oplossen', status: 'idle' }
  //   { step: 1, name: 'Secties samenvoegen', status: 'idle' }
  //   { step: 2, name: 'DOCX genereren', status: 'idle' }
  //   { step: 3, name: 'Diagrammen renderen', status: 'idle' }
  //
  // start(options: { mode: string, language: string }):
  //   - Reset all stages to idle
  //   - Set isRunning = true
  //   - Create EventSource to /api/projects/${projectId}/export/stream?mode=${options.mode}&language=${options.language}
  //   - Listen for events: step_start, step_done, complete, error, cancelled
  //   - step_start: set stage[step].status = 'running'
  //   - step_done: set stage[step].status = 'done'
  //   - complete: set isRunning = false, close EventSource. Store artifact_filename.
  //   - error: set stage[step].status = 'error', stage[step].errorMessage = message, close EventSource, set isRunning = false
  //   - onerror: set isRunning = false, close EventSource
  //
  // cancel():
  //   - eventSourceRef.current?.close()
  //   - Set isRunning = false
  //   - Reset stages to idle
  //
  // Return { stages, isRunning, start, cancel, completedFilename }
}
```
IMPORTANT: The EventSource URL must NOT use the `/api` prefix from the api client — it's a direct browser connection. Use the full path: `/api/projects/${projectId}/export/stream?...` (Vite proxy forwards /api to backend).
