      content: string

      options?: string[]
      selected_topics?: string[]
      decision?: Decision
      attachments?: string[]
    }

    export interface StreamEvent {

      data: {

        accumulated?: string
        final?: string
        question?: string
        options?: string[]
        decision?: Decision
        error?: string
      }
    }

    **3. Create SSE streaming hook** at `frontend/src/features/discussions/hooks/useDiscussionStream.ts`:
    - Uses browser-native EventSource for SSE connection
    - State: messages array, isStreaming boolean, currentStreamedContent string, error
    - `startDiscussion(projectId, phaseNumber)` -- POST to create conversation, then open SSE
    - `sendMessage(content, attachments?)` -- POST to /messages/stream endpoint, handle SSE events
    - Event handlers:
      - `message_delta`: append chunk to currentStreamedContent, update last message
      - `message_complete`: finalize message, clear streaming state
      - `question_card`: add structured question message
      - `summary_card`: add summary message with decision data
      - `error`: set error state, show toast
      - `done`: close EventSource
    - Auto-scroll behavior: track whether user has scrolled up. If at bottom, auto-scroll. If scrolled up, show "Nieuw bericht" button.
    - NOTE: EventSource only supports GET. For POST with body, use fetch with ReadableStream instead:
      ```typescript
      const response = await fetch(`/api/projects/${projectId}/discussions/${conversationId}/messages/stream`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ content, attachments })
      })
      const reader = response.body?.getReader()
      const decoder = new TextDecoder()
      // Read SSE events from stream
      ```
    - Cleanup: abort controller to cancel stream on unmount
