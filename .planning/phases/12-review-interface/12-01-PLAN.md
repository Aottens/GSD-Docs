2. In `backend/app/api/phases.py`, add two new private functions:

`_parse_verification_summary_table(content: str) -> list[dict]`:
- Split content by lines, find the summary table (line starting with `| Truth` or between `## Summary` and next `##`)
- Parse each data row: extract truth description (col 1), level results (cols 2-6 mapping checkmark->pass, warning->gap, dash->'-', N/A->n/a), status (col 7: PASS/GAP)
- Return list of dicts: `{"description": str, "exists": str, "substantive": str, "complete": str, "consistent": str, "standards": str, "status": str}`
- Handle checkmark chars: both "✓" (U+2713) and "✓" (checkmark variations) map to "pass"; "⚠" maps to "gap"; "-" maps to "-"; "N/A" maps to "n/a"

`_parse_verification_detail(content: str) -> VerificationDetailResponse`:
- Extract **Status:** line: `re.search(r'\*\*Status:\*\*\s*(PASS|GAPS_FOUND(?:\s*\(ESCALATED\))?)', content, re.MULTILINE)` — default "UNKNOWN"
- Extract **Cycle:** line: `re.search(r'\*\*Cycle:\*\*\s*(\d+)\s+of\s+(\d+)', content, re.MULTILINE)` — default 1 of 2
- Call `_parse_verification_summary_table(content)` for quick per-truth status
- Split on `### Truth` pattern to get detailed finding blocks
- For each truth block:
  - Extract status line: `**Status:** PASS` or `**Status:** GAP (Level N - Level Name)`
  - Extract gap description: text between `**Gap description:**` and `**Evidence:**` (or next `---`)
  - Extract evidence file paths: lines matching `- File: {path}` under `**Evidence:**`
  - Extract standards violations: regex `(PackML|ISA-88)\s*[§#]?[\d.]+` from gap description text; each match becomes `{"reference": matched_text, "text": gap_description_paragraph_containing_it}`
  - Map summary table row to detailed finding by matching truth description
- Build TruthResult list combining summary + detail
- Return VerificationDetailResponse with computed totals

```tsx
import { createContext, useContext, useReducer, useEffect, useCallback, useMemo } from 'react'
import type { ReactNode } from 'react'
import type { SectionReview, ReviewContextValue } from '../types/verification'

const STORAGE_KEY = (projectId: number, phaseNumber: number) =>
  `review-session-${projectId}-${phaseNumber}`

type ReviewAction =
  | { type: 'SET_REVIEW'; sectionId: string; status: SectionReview['status']; feedback: string }
  | { type: 'CLEAR' }
  | { type: 'LOAD'; reviews: Record<string, SectionReview> }

function reviewReducer(
  state: Record<string, SectionReview>,
  action: ReviewAction
): Record<string, SectionReview> {
  switch (action.type) {
    case 'SET_REVIEW':
      return {
        ...state,
        [action.sectionId]: {
          sectionId: action.sectionId,
          status: action.status,
          feedback: action.feedback,
          timestamp: Date.now(),
        },
      }
    case 'CLEAR':
      return {}
    case 'LOAD':
      return action.reviews
    default:
      return state
  }
}

const ReviewContext = createContext<ReviewContextValue | null>(null)

interface ReviewProviderProps {
  projectId: number
  phaseNumber: number
  children: ReactNode
}

export function ReviewProvider({ projectId, phaseNumber, children }: ReviewProviderProps) {
  const storageKey = STORAGE_KEY(projectId, phaseNumber)

  const [reviews, dispatch] = useReducer(reviewReducer, {}, () => {
    try {
      const stored = localStorage.getItem(storageKey)
      return stored ? JSON.parse(stored) : {}
    } catch {
      return {}
    }
  })

  // Persist on change
  useEffect(() => {
    localStorage.setItem(storageKey, JSON.stringify(reviews))
  }, [reviews, storageKey])

  const setReview = useCallback(
    (sectionId: string, status: SectionReview['status'], feedback: string) => {
      dispatch({ type: 'SET_REVIEW', sectionId, status, feedback })
    },
    []
  )

  const clearReviews = useCallback(() => {
    dispatch({ type: 'CLEAR' })
  }, [])

  const exportAsJson = useCallback(() => {
    return JSON.stringify(
      {
        phaseNumber,
        exportedAt: new Date().toISOString(),
        sections: Object.values(reviews).map((r) => ({
          section_id: r.sectionId,
          status: r.status === 'goedgekeurd' ? 'Approved' : r.status === 'opmerking' ? 'Comment' : 'Flag',
          finding: r.feedback || '-',
        })),
      },
      null,
      2
    )
  }, [phaseNumber, reviews])

  const reviewedCount = useMemo(() => Object.keys(reviews).length, [reviews])

  const value = useMemo<ReviewContextValue>(
    () => ({ reviews, setReview, clearReviews, exportAsJson, reviewedCount }),
    [reviews, setReview, clearReviews, exportAsJson, reviewedCount]
  )

  return <ReviewContext.Provider value={value}>{children}</ReviewContext.Provider>
}

export function useReviewContext(): ReviewContextValue {
  const ctx = useContext(ReviewContext)
  if (!ctx) {
    throw new Error('useReviewContext must be used within a ReviewProvider')
  }
  return ctx
}
```

Key details:
- `exportAsJson()` maps Dutch status names to REVIEW.md template format: goedgekeurd->Approved, opmerking->Comment, afgewezen->Flag
- localStorage key format: `review-session-{projectId}-{phaseNumber}`
- Lazy initializer on useReducer reads from localStorage on mount
- useEffect persists to localStorage on every reviews state change
